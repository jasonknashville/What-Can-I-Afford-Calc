function calculate(){ let income = document.getElementById("income").value / 12; let car = parseFloat(document.getElementById("car").value) || 0; let cc = parseFloat(document.getElementById("cc").value) || 0; let other = parseFloat(document.getElementById("other").value) || 0; let debts = car + cc + other; let rate = document.getElementById("rate").value / 100 / 12; let term = document.getElementById("term").value * 12; let down = document.getElementById("down").value / 100; let taxRate = document.getElementById("county").value; let insurance = parseFloat(document.getElementById("insurance").value) || 150; let extra = parseFloat(document.getElementById("extra").value) || 0; /* DTI limits */ let front = income * .36; let back = (income * .45) - debts; let maxHousing = Math.min(front, back); if(maxHousing <= 0){ document.getElementById("result").innerHTML = "Based on these inputs, debts exceed typical lender qualification limits."; return; } /* Estimate price by solving PITI */ let estimatedTax = 300; // starting estimate let availablePI = maxHousing - insurance - estimatedTax; let loan = availablePI * ((1 - Math.pow(1 + rate, -term)) / rate); let price = loan / (1 - down); /* refine tax estimate */ estimatedTax = (price * taxRate) / 12; availablePI = maxHousing - insurance - estimatedTax; loan = availablePI * ((1 - Math.pow(1 + rate, -term)) / rate); price = loan / (1 - down) + extra; if(price < 0){ price = 0; } document.getElementById("result").innerHTML = "Estimated Maximum Home Price:
$" + Math.round(price).toLocaleString() + ""; }
